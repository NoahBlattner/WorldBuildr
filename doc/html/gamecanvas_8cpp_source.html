181bool GameCanvas::eventFilter(QObject* pWatched, QEvent* pEvent)
182{
183 // Les événements claviers doivent être traités au niveau de GameView, sinon, certaines touches
184 // auront un effet indésirable pris en charge par QGraphicsView (ou plus exactement QAbstractScrollBar, qui
185 // gère le comportement des touches fléchées et de PageUp et PageDown).
186 // Pour que l'événement ne soit plus traité plus loin (et donc soit absorbé ici), on retourne true, comme
187 // expliqué dans la doc de QObject::eventFilter() :
188 //
189 // "In your reimplementation of this function, if you want to filter the event out, i.e. stop it being handled
190 // further, return true; otherwise return false."
191 //
192 if (qobject_cast<QGraphicsView*>(pWatched)) {
193 switch (pEvent->type()) {
194 case QEvent::KeyPress: this->keyPressed(static_cast<QKeyEvent*>(pEvent)); return true;
195 case QEvent::KeyRelease: this->keyReleased(static_cast<QKeyEvent *>(pEvent)); return true;
196 default : return QObject::eventFilter(pWatched, pEvent);
197 }
198 }
199
200 // Les événements restants proviennent de GameScene.
201 // Attention : les événements de la souris, une fois traités par GameCanevas, doivent pouvoir être gérés plus
202 // loin, par exemple dans le cas où la scène intègre des QWidget. Il faut donc retourner false.
203 switch (pEvent->type()) {
204 case QEvent::GraphicsSceneMouseMove: this->mouseMoved(static_cast<QGraphicsSceneMouseEvent*>(pEvent)); return false;
205 case QEvent::GraphicsSceneMousePress: this->mouseButtonPressed(static_cast<QGraphicsSceneMouseEvent*>(pEvent)); return false;
206 case QEvent::GraphicsSceneMouseRelease: this->mouseButtonReleased(static_cast<QGraphicsSceneMouseEvent*>(pEvent)); return false;
207 default : return QObject::eventFilter(pWatched, pEvent);
208 }
209}

226void GameCanvas::keyPressed(QKeyEvent* pKeyEvent) {
227 // Supprimer ce premier test si la répétition de touche doit être signalée.
228 if (pKeyEvent->isAutoRepeat())
229 pKeyEvent->ignore();
230 else {
231 m_pGameCore->keyPressed(pKeyEvent->key());
232
233 if (pKeyEvent->modifiers()==(Qt::ShiftModifier|Qt::ControlModifier)) {
234 switch (pKeyEvent->key()) {
235 case Qt::Key_I:
236 if (m_pDetailedInfosItem)
237 m_pDetailedInfosItem->setVisible(!m_pDetailedInfosItem->isVisible());
238 break;
239 case Qt::Key_P:
240 m_tickTimer.setInterval(m_tickTimer.interval()+1);
241 qDebug() << "Tick interval set to " << m_tickTimer.interval();
242 break;
243 case Qt::Key_M:
244 m_tickTimer.setInterval(m_tickTimer.interval()-1);
245 qDebug() << "Tick interval set to " << m_tickTimer.interval();
246 break;
247 }
248 }
249 pKeyEvent->accept();
250 }
251}

301void GameCanvas::onTick() {
302 long long elapsedTime = m_lastUpdateTime.elapsed();
303
304 // On évite une division par zéro (peu probable, mais on sait jamais)
305 if (elapsedTime < 1)
306 elapsedTime = 1;
307
308 m_lastUpdateTime.start();
309
310 // Statistiques
311 m_tickCount++;
312 m_totalElapsedTime += elapsedTime;
313
314 // Tick
315 m_pGameCore->tick(elapsedTime);
316 currentScene()->tick(elapsedTime);
317
318 if (m_pDetailedInfosItem && m_pDetailedInfosItem->isVisible())
319 m_pDetailedInfosItem->setPlainText(QString("FPS : %1, Elapsed : %2ms, Tick duration : %3ms")
320 .arg(1000/elapsedTime)
321 .arg(elapsedTime)
322 .arg(m_lastUpdateTime.elapsed()));
323
324#ifdef QT_DEBUG
325 // Statistiques
326 m_statsTrigger -= elapsedTime;
327 if (m_statsTrigger < 0) {
328 //qDebug() << "Tick count :" << m_tickCount << ". Mean tick duration : " << m_totalElapsedTime / m_tickCount << ". Min duration : " << m_minTickDuration << ". Max duration : " << m_maxTickDuration;
329 resetStatistics();
330 }
331
332 if (m_lastUpdateTime.elapsed() > m_maxTickDuration)
333 m_maxTickDuration = m_lastUpdateTime.elapsed();
334 else if (m_lastUpdateTime.elapsed() < m_minTickDuration)
335 m_minTickDuration = m_lastUpdateTime.elapsed();
336#endif
