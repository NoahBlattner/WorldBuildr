386void Sprite::onNextAnimationFrame() {
387 if (m_animationList[m_currentAnimationIndex].isEmpty()) {
388 m_currentAnimationFrame = NO_CURRENT_FRAME;
389 return;
390 }
391
392 int PreviousAnimationFrame = m_currentAnimationFrame;
393 ++m_currentAnimationFrame;
394 if (m_currentAnimationFrame >= m_animationList[m_currentAnimationIndex].count()) {
395 m_currentAnimationFrame = 0;
396 if (m_emitSignalEOA)
397 emit animationFinished();
398 if (m_animationStopLater) {
399 m_animationStopLater = false;
400 stopAnimation(IMMEDIATE_STOP);
401 }
402 }
403 if (PreviousAnimationFrame != m_currentAnimationFrame) {
404 setPixmap(m_animationList[m_currentAnimationIndex][m_currentAnimationFrame]);
405 update();
406 }
407}
